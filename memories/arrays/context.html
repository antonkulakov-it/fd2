Перебор с помощью for...in

            var divs = document.getElementsByTagName('div');
            for(i in divs) {
                console.log(divs[i]);
            }
Плохой метод - выводит не нужные свойства.

Перебор с помощью цикла

            for(var i=0; i < divs.length; i++) {
                console.log(divs[i]);
            }
Может быть применен в большинстве случаев.

Перебор с помощью переназначения контекста

            [].forEach.call(divs, function(p,i,arr) {
                console.log(p, i, arr[i]);
            });
            
            [].forEach.apply(divs, [function(p,i,arr) {
                console.log(p, i, arr[i]);
            }]);
